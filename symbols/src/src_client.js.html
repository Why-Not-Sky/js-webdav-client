 26 /**
 27  * @class Connection to a WebDAV server
 28  *
 29  * @param  {Array}  [config]  A configuration object. It can contain the following fields:
 30  *                            - host; String containing the hostname to connect to (default: current host)
 31  *                            - useHTTPS; If set to (boolean) false, a regular HTTP connection will be used, any other value (even 0 or '' which evaluate to false) and HTTPS will be used instead. Is ignored if host is not set. (default: true)
 32  *                            - port; Integer with the port number to use. Is ignored if host is not set. (default: 443 for HTTPS and 80 for HTTP)
 33  *                            - defaultHeaders; An array containing default headers to use for each request. Header names should be the keys. (default: {} i.e. no default headers)
 34  *                            - username; A string with the username to use for authentication. (default: the current username if any)
 35  *                            - password; A string with the password to use for authentication. This is only used if a username is supplied. (default: the current password if any)
 36  *                            For backwards compatibility reasons, it is also
 37  *                            possible to specify the 'host', 'useHTTPS', 'port'
 38  *                            and 'defaultHeaders' as regular parameters (in
 39  *                            that order). Note that useHTTPS works differently
 40  *                            in that case; any other value than boolean True
 41  *                            will result in regular HTTP. If you want to supply
 42  *                            a username and password, you will have to use the
 43  *                            config array, you cannot pass them as extra normal
 44  *                            parameters.
 45  */
 46 nl.sara.webdav.Client = function(config, useHTTPS, port, defaultHeaders) {
 47   // First define private attributes
 48   Object.defineProperty(this, '_baseUrl', {
 49     'value': null,
 50     'enumerable': false,
 51     'configurable': false,
 52     'writable': true
 53   });
 54   Object.defineProperty(this, '_username', {
 55     'value': null,
 56     'enumerable': false,
 57     'configurable': false,
 58     'writable': true
 59   });
 60   Object.defineProperty(this, '_password', {
 61     'value': null,
 62     'enumerable': false,
 63     'configurable': false,
 64     'writable': true
 65   });
 66   Object.defineProperty(this, '_headers', {
 67     'value': {},
 68     'enumerable': false,
 69     'configurable': false,
 70     'writable': true
 71   });
 72 
 73   // Constructor logic
 74   if ( config !== undefined ) {
 75     var host;
 76     if ( typeof( config ) !== 'string' ) {
 77       host = config['host'];
 78       useHTTPS = ( config['useHTTPS'] !== false );
 79       port = config['port'];
 80       defaultHeaders = config['defaultHeaders'];
 81       
 82       if ( config['username'] !== undefined ) {
 83         this._username = config['username'];
 84         if ( config['password'] !== undefined ) {
 85           this._password = config['password'];
 86         }else{
 87           this._password = '';
 88         }
 89       }
 90     }else{
 91       host = config;
 92     }
 93     
 94     // if the configuration item is a string, then we have to work in compatibility mode; first parameter is the host, second, the protocol, third the port and fourth aditional headers
 95     var protocol = (useHTTPS === true) ? 'https' : 'http';
 96     port = (port !== undefined) ? port : ((protocol === 'https') ? 443 : 80);
 97     this._baseUrl = protocol + '://' + host + ((((protocol === 'http') && (port === 80)) || ((protocol === 'https') && (port === 443))) ? '' : ':' + port);
 98   }
 99   
100   if (defaultHeaders !== undefined) {
101     this._headers = defaultHeaders;
102   }
103 };

117 /**
118  * Converts a path to the full url (i.e. appends the protocol and host part)
119  *
120  * @param   {String}  path  The path on the server
121  * @returns {String}        The full url to the path
122  */
123 nl.sara.webdav.Client.prototype.getUrl = function(path) {
124   if (path.substring(0,1) !== '/') {
125     path = '/' + path;
126   }
127   if (this._baseUrl !== null) {
128     return this._baseUrl + path;
129   }else{
130     return path;
131   }

134 /**
135  * Perform a WebDAV PROPFIND request
136  *
137  * @param    {String}                         path             The path get a PROPFIND for
138  * @param    {Function(status,body,headers)}  callback         Querying the server is done asynchronously, this callback function is called when the results are in
139  * @param    {String}                         [depth=0]        Optional; Value for the 'depth' header, should be either '0', '1' or the class constant INFINITY. When omitted, '0' is used. See RFC 4916.
140  * @param    {mixed}                          [props=ALLPROP]  Optional; The properties to fetch. Should be either one of the class constants PROPNAME or ALLPROP or an array with Property objects. When omitted, ALLPROP is assumed. See RFC 4916.
141  * @param    {nl.sara.webdav.Property[]}      [include]        Optional; An array with Property objects used for the <include> element. This is only used for ALLPROP requests. When omitted, no <include> element is send. See RFC 4916.
142  * @param    {Array}                          headers          Optional; Additional headers to set
143  * @returns  {nl.sara.webdav.Client}                           The client itself for chaining methods
144  */
145 nl.sara.webdav.Client.prototype.propfind = function(path, callback, depth, props, include, headers) {
146   if ((path === undefined) || (callback === undefined)) {
147     throw new nl.sara.webdav.Exception('PROPFIND requires the parameters path and callback', nl.sara.webdav.Exception.MISSING_REQUIRED_PARAMETER);
148   }
149   if (!(typeof path === "string")) {
150     throw new nl.sara.webdav.Exception('PROPFIND parameter; path should be a string', nl.sara.webdav.Exception.WRONG_TYPE);
151   }
152 
153   // Get the full URL, based on the specified path
154   var url = this.getUrl(path);
155 
156   // Check the depth header
157   if (depth === undefined) { // We default depth to 0, not to infinity as this is not supported by all servers
158     depth = 0;
159   }
160   var depthHeader = null;
161   switch (depth) {
162     case 0:
163     case 1:
164       depthHeader = depth;
165       break;
166     case nl.sara.webdav.Client.INFINITY:
167       depthHeader = 'infinity';
168       break;
169     default:
170       throw new nl.sara.webdav.Exception("Depth header should be '0', '1' or nl.sara.webdav.Client.INFINITY", nl.sara.webdav.Exception.WRONG_VALUE);
171     break;
172   }
173 
174   // Create the request XML
175   if (props === undefined) {
176     props = nl.sara.webdav.Client.ALLPROP;
177   }
178   var propsBody = document.implementation.createDocument("DAV:", "propfind", null);
179   switch (props) { // Find out what the request is for
180     case nl.sara.webdav.Client.PROPNAME: // User wants all property names
181       propsBody.documentElement.appendChild(propsBody.createElementNS('DAV:', 'propname'));
182       break;
183     case nl.sara.webdav.Client.ALLPROP: // User wants all properties
184       propsBody.documentElement.appendChild(propsBody.createElementNS('DAV:', 'allprop'));
185       if (include !== undefined) { // There is content for the <DAV: include> tags, so parse it
186         if (!(include instanceof Array)) {
187           throw new nl.sara.webdav.Exception('Propfind parameter; include should be an array', nl.sara.webdav.Exception.WRONG_TYPE);
188         }
189         var includeBody = propsBody.createElementNS('DAV:', 'include');
190         for (var i = 0; i < include.length; i++) {
191           var item = include[i];
192           if (!nl.sara.webdav.Ie.isIE && !(item instanceof nl.sara.webdav.Property)) {
193             continue;
194           }
195           includeBody.appendChild(propsBody.createElementNS(item.namespace, item.tagname));
196         }
197         if (includeBody.hasChildNodes()) { // But only really add the <include> tag if there is valid content
198           propsBody.documentElement.appendChild(includeBody);
199         }
200       }
201       break;
202     default: // The default is to expect an array with Property objects; the user wants the values of these properties
203       if (!(props instanceof Array)) {
204         throw new nl.sara.webdav.Exception('Props parameter should be nl.sara.webdav.Client.PROPNAME, nl.sara.webdav.Client.ALLPROP or an array with Property objects', nl.sara.webdav.Exception.WRONG_VALUE);
205       }
206       var propertyBody = propsBody.createElementNS('DAV:', 'prop');
207       for (var i = 0; i < props.length; i++) { // Cycle through the array
208         var prop = props[i];
209         if (!nl.sara.webdav.Ie.isIE && !(prop instanceof nl.sara.webdav.Property)) {
210           continue;
211         }
212         propertyBody.appendChild(propsBody.createElementNS(prop.namespace, prop.tagname));
213       }
214       if (!propertyBody.hasChildNodes()) { // But if no properties are found, then the array didn't have Property objects in it
215         throw new nl.sara.webdav.Exception("Propfind parameter; if props is an array, it should be an array of Properties", nl.sara.webdav.Exception.WRONG_TYPE);
216       }
217       propsBody.documentElement.appendChild(propertyBody);
218     break;
219   }
220   var serializer = new XMLSerializer();
221   var body = '<?xml version="1.0" encoding="utf-8" ?>' + serializer.serializeToString(propsBody);
222 
223   // And then send the request
224   var ajax = this.getAjax("PROPFIND", url, callback, headers);
225   ajax.setRequestHeader('Depth', depthHeader);
226   ajax.setRequestHeader('Content-Type', 'application/xml; charset="utf-8"');
227   ajax.send(body);
228 
229   return this;
230 };

232 /**
233  * Perform a WebDAV PROPPATCH request
234  *
235  * @param    {String}                         path        The path do a PROPPATCH on
236  * @param    {Function(status,body,headers)}  callback    Querying the server is done asynchronously, this callback function is called when the results are in
237  * @param    {nl.sara.webdav.Property[]}      [setProps]  Optional; The properties to set. If not set, delProps should be set. Can be omitted with 'undefined'.
238  * @param    {nl.sara.webdav.Property[]}      [delProps]  Optional; The properties to delete. If not set, setProps should be set.
239  * @param    {Array}                          headers     Optional; Additional headers to set
240  * @returns  {nl.sara.webdav.Client}                      The client itself for chaining methods
241  */
242 nl.sara.webdav.Client.prototype.proppatch = function(path, callback, setProps, delProps, headers) {
243   if ((path === undefined) || (callback === undefined) || ((setProps === undefined) && (delProps === undefined))) {
244     throw new nl.sara.webdav.Exception('PROPPATCH requires the parameters path, callback and at least one of setProps or delProps', nl.sara.webdav.Exception.MISSING_REQUIRED_PARAMETER);
245   }
246   if (!(typeof path === "string") || ((setProps !== undefined) && !(setProps instanceof Array)) || ((delProps !== undefined) && !(delProps instanceof Array))) {
247     throw new nl.sara.webdav.Exception('PROPPATCH parameter; path should be a string, setProps and delProps should be arrays', nl.sara.webdav.Exception.WRONG_TYPE);
248   }
249 
250   // Get the full URL, based on the specified path
251   var url = this.getUrl(path);
252 
253   // Create the request XML
254   var propsBody = document.implementation.createDocument("DAV:", "propertyupdate", null);
255   propsBody.documentElement.setAttribute("xmlns:D", "DAV:");
256   if (setProps !== undefined) {
257     var props = propsBody.createElementNS('DAV:', 'prop');
258     for (var i = 0; i < setProps.length; i++) { // Cycle through the array
259       var prop = setProps[i];
260       if (!nl.sara.webdav.Ie.isIE && !(prop instanceof nl.sara.webdav.Property)) {
261         continue;
262       }
263       var element = propsBody.createElementNS(prop.namespace, prop.tagname);
264       for (var j = 0; j < prop.xmlvalue.length; j++) {
265         var nodeCopy = propsBody.importNode(prop.xmlvalue.item(j), true);
266         element.appendChild(nodeCopy);
267       }
268       props.appendChild(element);
269     }
270     if (!props.hasChildNodes()) { // But if no properties are found, then the array didn't have Property objects in it
271       throw new nl.sara.webdav.Exception("Proppatch parameter; setProps should be an array of Properties", nl.sara.webdav.Exception.WRONG_TYPE);
272     }
273     var set = propsBody.createElementNS('DAV:', 'set');
274     set.appendChild(props);
275     propsBody.documentElement.appendChild(set);
276   }
277   if (delProps !== undefined) {
278     var props = propsBody.createElementNS('DAV:', 'prop');
279     for (var i = 0; i < delProps.length; i++) { // Cycle through the array
280       var prop = delProps[i];
281       if (!nl.sara.webdav.Ie.isIE && !(prop instanceof nl.sara.webdav.Property)) {
282         continue;
283       }
284       var element = propsBody.createElementNS(prop.namespace, prop.tagname);
285       props.appendChild(element);
286     }
287     if (!props.hasChildNodes()) { // But if no properties are found, then the array didn't have Property objects in it
288       throw new nl.sara.webdav.Exception("Proppatch parameter; delProps should be an array of Properties", nl.sara.webdav.Exception.WRONG_TYPE);
289     }
290     var remove = propsBody.createElementNS('DAV:', 'remove');
291     remove.appendChild(props);
292     propsBody.documentElement.appendChild(remove);
293   }
294   var serializer = new XMLSerializer();
295   var body = '<?xml version="1.0" encoding="utf-8" ?>' + serializer.serializeToString(propsBody);
296 
297   // And then send the request
298   var ajax = this.getAjax("PROPPATCH", url, callback, headers);
299   ajax.setRequestHeader('Content-Type', 'application/xml; charset="utf-8"');
300   ajax.send(body);
301 
302   return this;
303 };

494 /**
495  * Perform a WebDAV COPY request
496  *
497  * @param    {String}                         path                              The path to perform COPY on
498  * @param    {Function(status,body,headers)}  callback                          Querying the server is done asynchronously, this callback function is called when the results are in
499  * @param    {String}                         destination                       The destination to copy to. Should be either a full URL or a path from the root on this server (i.e. it should start with a /)
500  * @param    {Integer}                        [overwriteMode=SILENT_OVERWRITE]  Optional; Specify what to do when destination resource already exists. Should be either FAIL_ON_OVERWRITE or SILENT_OVERWRITE. The default is SILENT_OVERWRITE.
501  * @param    {String}                         [depth]                           Optional; Should be '0' or 'infinity'. This is used in case of a collection; 0 means only copy the collection itself, infinity means copy also everything contained in the collection
502  * @param    {Array}                          headers                           Optional; Additional headers to set
503  * @returns  {nl.sara.webdav.Client}                                            The client itself for chaining methods
504  */
505 nl.sara.webdav.Client.prototype.copy = function(path, callback, destination, overwriteMode, depth, headers) {
506   if ((path === undefined) || (callback === undefined) || (destination === undefined)) {
507     throw new nl.sara.webdav.Exception('COPY requires the parameters path, callback and destination', nl.sara.webdav.Exception.MISSING_REQUIRED_PARAMETER);
508   }
509   if ((typeof path !== "string") || (typeof destination !== "string")){
510     throw new nl.sara.webdav.Exception('COPY parameter; path and destination should be strings', nl.sara.webdav.Exception.WRONG_TYPE);
511   }
512 
513   // Get the full URL, based on the specified path
514   var url = this.getUrl(path);
515 
516   // If the destination starts with a / it is a absolute url on the same host, so prepare a complete URL
517   if (destination.substr(0,1) === '/') {
518     destination = this.getUrl(destination);
519   } // Else I assume it is a complete URL already
520 
521   // And then send the request
522   var ajax = this.getAjax("COPY", url, callback, headers);
523   ajax.setRequestHeader('Destination', destination);
524   if (depth !== undefined) {
525     if ((depth !== 0) && (depth !== 'infinity')) {
526       throw new nl.sara.webdav.Exception("COPY parameter; depth should be '0' or 'infinity'", nl.sara.webdav.Exception.WRONG_VALUE);
527     }
528     ajax.setRequestHeader('Depth', depth);
529   }
530   if (overwriteMode === nl.sara.webdav.Client.FAIL_ON_OVERWRITE) {
531     ajax.setRequestHeader('Overwrite', 'F');
532   }
533   ajax.send();
534 
535   return this;
536 };

603 /**
604  * Prepares a XMLHttpRequest object to be used for an AJAX request
605  *
606  * @static
607  * @param    {String}                         method    The HTTP/webDAV method to use (e.g. GET, POST, PROPFIND)
608  * @param    {String}                         url       The url to connect to
609  * @param    {Function(status,body,headers)}  callback  Querying the server is done asynchronously, this callback function is called when the results are in
610  * @param    {Array}                          headers   Additional headers to set
611  * @returns  {XMLHttpRequest}                           A prepared XMLHttpRequest
612  */
613 nl.sara.webdav.Client.prototype.getAjax = function(method, url, callback, headers) {
614   var /** @type XMLHttpRequest */ ajax = new XMLHttpRequest();
615   if ( this._username !== null ) {
616     ajax.open( method, url, true, this._username, this._password );
617   }else{
618     ajax.open( method, url, true );
619   }
620   ajax.onreadystatechange = function() {
621     nl.sara.webdav.Client.ajaxHandler( ajax, callback );
622   };
623   
624   if (headers === undefined) {
625     headers = {};
626   }
627   for (var header in this._headers) {
628     if (headers[header] === undefined) {
629       ajax.setRequestHeader(header, this._headers[header]);
630     }
631   }
632   for (var header in headers) {
633     ajax.setRequestHeader(header, headers[header]);
634   }
635   return ajax;
636 };

638 /**
639  * AJAX request handler. Parses Multistatus (if available) and call a user specified callback function
640  *
641  * @static
642  * @param    {XMLHttpRequest}                 ajax      The XMLHttpRequest object which performed the request
643  * @param    {Function(status,body,headers)}  callback  Querying the server is done asynchronously, this callback function is called when the results are in
644  * @returns  {void}
645  */
646 nl.sara.webdav.Client.ajaxHandler = function(ajax, callback) {
647   if (ajax.readyState === 4){ //if request has completed
648     var body = ajax.responseText;
649     if (ajax.status === 207) {
650       // Parse the response to a Multistatus object
651       for (var counter = 0; counter < ajax.responseXML.childNodes.length; counter++) {
652         if (nl.sara.webdav.Ie.getLocalName(ajax.responseXML.childNodes.item(counter)) === 'multistatus') {
653           body = new nl.sara.webdav.Multistatus(ajax.responseXML.childNodes.item(counter));
654           break;
655         }
656       }
657     }
658     callback(ajax.status, body, ajax.getAllResponseHeaders());
659   }
660 };
