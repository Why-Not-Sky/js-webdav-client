 21 // If nl.sara.webdav.codec.OwnerCodec is already defined, we have a namespace clash!
 22 if (nl.sara.webdav.codec.OwnerCodec !== undefined) {
 23   throw new nl.sara.webdav.Exception('Namespace nl.sara.webdav.codec.OwnerCodec already taken, could not load JavaScript library for WebDAV connectivity.', nl.sara.webdav.Exception.NAMESPACE_TAKEN);
 24 }
 25 
 26 /**
 27  * @class Adds a codec that converts DAV: owner to a Date object
 28  * @augments nl.sara.webdav.Codec
 29  */
 30 nl.sara.webdav.codec.OwnerCodec = new nl.sara.webdav.Codec();
 31 nl.sara.webdav.codec.OwnerCodec.namespace = 'DAV:';
 32 nl.sara.webdav.codec.OwnerCodec.tagname = 'owner';
 33 
 34 nl.sara.webdav.codec.OwnerCodec.fromXML = function(nodelist) {
 35   var returnValue = null;
 36 
 37   // Assertions whether the formed XML is correct
 38   var hrefNode = nodelist[0];
 39 
 40   if ((hrefNode.nodeType === 1) &&
 41       (hrefNode.namespaceURI === 'DAV:') &&
 42       (hrefNode.localName === 'href'))
 43   {
 44     var node = hrefNode.childNodes.item(0);
 45 
 46     if ( ( node.nodeType === 3 ) || ( node.nodeType === 4 ) ) { // Make sure text and CDATA content is stored
 47       returnValue = node.nodeValue;
 48     }
 49   } 
 50   return returnValue;
 51 };
 52 
 53 nl.sara.webdav.codec.OwnerCodec.toXML = function(value, xmlDoc){  
 54   var hrefNode = xmlDoc.createElementNS( 'DAV:', 'href' );
 55   hrefNode.appendChild( xmlDoc.createCDATASection( value.toString() ) );
 56   xmlDoc.documentElement.appendChild( hrefNode );
 57   
 58   return xmlDoc
 59 };
 60 
 61 nl.sara.webdav.Property.addCodec(nl.sara.webdav.codec.OwnerCodec);
